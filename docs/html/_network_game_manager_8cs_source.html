156 public async Task SendShipPlacementAsync(GameBoard playerBoard)
157 {
158 _playerBoard = playerBoard;
159 _localShipsPlaced = true;
160
161 // Отправляем расположение кораблей
162 try
163 {
164 var shipData = new Dictionary<string, string>();
165 for (int i = 0; i < playerBoard.Ships.Count; i++)
166 {
167 var ship = playerBoard.Ships[i];
168 shipData[$"ship{i}_size"] = ship.Size.ToString();
169 shipData[$"ship{i}_horizontal"] = ship.IsHorizontal.ToString();
170 shipData[$"ship{i}_positions"] = string.Join(",", ship.Positions.Select(p => $"{p.X}:{p.Y}"));
171 }
172
173 var placementMsg = new NetworkMessage
174 {
175 Type = NetworkProtocol.Commands.ShipPlacement,
176 Data = shipData
177 };
178 await _networkClient.SendMessageAsync(placementMsg);
179
180 // Сообщаем, что расстановка завершена
181 var readyMsg = new NetworkMessage { Type = NetworkProtocol.Commands.AllShipsPlaced };
182 await _networkClient.SendMessageAsync(readyMsg);
183 Console.WriteLine($"[Network] Отправлена информация о {playerBoard.Ships.Count} кораблях");
184 }
185 catch (Exception ex)
186 {
187 Console.WriteLine($"[Network] Ошибка отправки размещения кораблей: {ex.Message}");
188 throw;
189 }

295 private void OnNetworkMessageReceived(NetworkMessage message)

303 private void ProcessNetworkMessage(NetworkMessage message)
304 {
305 switch (message.Type.ToUpper())
306 {
307 case NetworkProtocol.Commands.Joined:
308 HandleJoinedMessage(message);
309 break;
310 case NetworkProtocol.Commands.MatchFound:
311 HandleMatchFoundMessage(message);
312 break;
313 case NetworkProtocol.Commands.GameStart:
314 HandleGameStartMessage(message);
315 break;
316 case NetworkProtocol.Commands.YourTurn:
317 HandleYourTurnMessage();
318 break;
319 case NetworkProtocol.Commands.YourTurnAgain:
320 HandleYourTurnAgainMessage();
321 break;
322 case NetworkProtocol.Commands.OpponentTurn:
323 HandleOpponentTurnMessage();
324 break;
325 case NetworkProtocol.Commands.AttackResult:
326 HandleAttackResultMessage(message);
327 break;
328 case NetworkProtocol.Commands.GameOver:
329 HandleGameOverMessage(message);
330 break;
331 case NetworkProtocol.Commands.OpponentLeft:
332 HandleOpponentLeftMessage(message);
333 break;
334 case NetworkProtocol.Commands.OpponentDisconnected:
335 HandleOpponentDisconnectedMessage();
336 break;
337 case NetworkProtocol.Commands.ChatMessageReceived:
338 HandleChatMessage(message.Data);
339 break;
340 case NetworkProtocol.Commands.Error:
341 HandleErrorMessage(message);
342 break;
343 default:
344 Console.WriteLine($"[Network] Неизвестная команда: {message.Type}");
345 break;
346 }
347 }

459 private void HandleAttackResultMessage(NetworkMessage message)
460 {
461 Console.WriteLine("[DEBUG] ATTACK_RESULT received");
462
463 if (int.TryParse(message.Data.GetValueOrDefault(NetworkProtocol.Keys.X, ""), out int x) &&
464 int.TryParse(message.Data.GetValueOrDefault(NetworkProtocol.Keys.Y, ""), out int y))
465 {
466 bool hit = message.Data.GetValueOrDefault(NetworkProtocol.Keys.Hit, "false") == "true";
467 bool sunk = message.Data.GetValueOrDefault(NetworkProtocol.Keys.Sunk, "false") == "true";
468 bool gameOver = message.Data.GetValueOrDefault(NetworkProtocol.Keys.GameOver, "false") == "true";
469 var attackerId = message.Data.GetValueOrDefault(NetworkProtocol.Keys.AttackerId, "");
470 bool isMyAttack = attackerId == _myPlayerId;
471
472 Console.WriteLine($"[DEBUG] ATTACK_RESULT: ({x},{y}), hit={hit}, sunk={sunk}, gameOver={gameOver}, isMyAttack={isMyAttack}");
473
474 // Снимаем блокировку атаки
475 _isProcessingNetworkAttack = false;
476
477 // Передаем все данные для обработки в MainWindow
478 AttackResultReceived?.Invoke(x, y, hit, sunk, gameOver, isMyAttack, message.Data);
479 }
480 else
481 {
482 Console.WriteLine("[DEBUG] Failed to parse ATTACK_RESULT coordinates");
483 }
