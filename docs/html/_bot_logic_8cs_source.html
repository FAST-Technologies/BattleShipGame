41 public async Task<BotTurnResult> MakeSimpleTurn(GameBoard playerBoard, Action<int, int, bool, bool, bool> onAttack)
42 {
43 var possibleTargets = GetSmartTargets(playerBoard);
44
45 if (possibleTargets.Count == 0)

70 public async Task<BotTurnResult> MakeSmartTurn(GameBoard playerBoard, Action<int, int, bool, bool, bool> onAttack)
71 {
72 var possibleTargets = GetSmartTargets(playerBoard);
73
74 if (possibleTargets.Count == 0)
75 return new BotTurnResult(false, false, false);
76
77 var target = GetNextSmartShot(possibleTargets, playerBoard);
78 possibleTargets.Remove((target.x, target.y));
79
80 var (hit, sunk, gameOver) = playerBoard.Attack(target.x, target.y);
81
82 // Сохраняем информацию о попадании для умного ИИ
83 if (hit)
84 {
85 _lastHits.Add((target.x, target.y));
86 if (_lastHits.Count > 5) _lastHits.RemoveAt(0);
87
88 _initialHit ??= (target.x, target.y);
89
90 // Определяем направление при втором попадании
91 if (_lastHits.Count >= 2)
92 {
93 var last = _lastHits[^1];
94 var prev = _lastHits[^2];
95 _lastHitDirection = (last.x - prev.x, last.y - prev.y);
96 }
97
98 if (sunk)
99 {
100 // Сбрасываем информацию при потоплении
101 _lastHits.Clear();
102 _lastHitDirection = null;
103 _initialHit = null;
104 }
105 }
106
107 // Вызываем колбэк для обработки результатов
108 onAttack?.Invoke(target.x, target.y, hit, sunk, gameOver);
109
110 // Определяем, продолжать ли ход
111 bool continueTurn = hit && !gameOver;
112 bool isPlayerTurn = !continueTurn && !gameOver;
113
114 return new BotTurnResult(continueTurn, isPlayerTurn, gameOver);
115 }

122 private List<(int x, int y)> GetSmartTargets(GameBoard board)
123 {
124 var targets = new List<(int x, int y)>();
125 for (int x = 0; x < board.Size; x++)
126 for (int y = 0; y < board.Size; y++)
127 if (board.Grid[x, y] == CellState.Empty || board.Grid[x, y] == CellState.Ship)
128 targets.Add((x, y));
129 return targets;

139 private (int x, int y) GetNextSmartShot(List<(int x, int y)> possibleTargets, GameBoard board)
140 {
141 switch (_difficulty)
142 {
143 // === СЛОЖНЫЙ ИИ: стреляем рядом с попаданиями ===
144 case BotDifficulty.Hard:
145 return GetSmartTargetHard(possibleTargets, board);
146
147 // === СРЕДНИЙ: приоритет соседям попаданий ===
148 case BotDifficulty.Medium:
149 return GetSmartTargetMedium(possibleTargets, board);
150
151 // === ЛЁГКИЙ: рандом ===
152 case BotDifficulty.Easy:
153 default:
154 return GetRandomTarget(possibleTargets);
155 }

164 private (int x, int y) GetSmartTargetHard(List<(int x, int y)> possibleTargets, GameBoard board)
165 {
166 if (_lastHits.Count == 0)
167 return GetRandomTarget(possibleTargets);
168
169 var lastHit = _lastHits.Last();
170
171 // 1. Продолжаем в том же направлении
172 if (_lastHitDirection.HasValue && _lastHits.Count >= 2)
173 {
174 var (dx, dy) = _lastHitDirection.Value;
175 int nextX = lastHit.x + dx;
176 int nextY = lastHit.y + dy;
177
178 if (IsValidAndAvailable(nextX, nextY, possibleTargets, board))
179 return (nextX, nextY);
180 }
181
182 // 2. Пробуем противоположное направление
183 if (_lastHitDirection.HasValue && _initialHit.HasValue)
184 {
185 var (dx, dy) = _lastHitDirection.Value;
186 int oppositeX = _initialHit.Value.x - dx;
187 int oppositeY = _initialHit.Value.y - dy;
188
189 if (IsValidAndAvailable(oppositeX, oppositeY, possibleTargets, board))
190 return (oppositeX, oppositeY);
191 }
192
193 // 3. Стреляем по соседям последнего попадания
194 var neighbors = GetNeighbors(lastHit.x, lastHit.y, board)
195 .Where(neighbor => possibleTargets.Contains(neighbor))
196 .ToList();
197
198 if (neighbors.Count > 0)
199 return neighbors[_random.Next(neighbors.Count)];
200
201 return GetRandomTarget(possibleTargets);
202 }

210 private (int x, int y) GetSmartTargetMedium(List<(int x, int y)> possibleTargets, GameBoard board)
211 {
212 if (_lastHits.Count == 0)
213 return GetRandomTarget(possibleTargets);
214
215 // Собираем всех соседей всех попаданий
216 var allNeighbors = new List<(int x, int y)>();
217 foreach (var hit in _lastHits)
218 allNeighbors.AddRange(GetNeighbors(hit.x, hit.y, board));
219
220 // Убираем дубликаты и выбираем только доступные цели
221 var uniqueNeighbors = allNeighbors
222 .Distinct()
223 .Where(neighbor => possibleTargets.Contains(neighbor))
224 .ToList();
225
226 if (uniqueNeighbors.Count > 0)
227 return uniqueNeighbors[_random.Next(uniqueNeighbors.Count)];
228
229 return GetRandomTarget(possibleTargets);
230 }

254 private bool IsValidAndAvailable(int x, int y, List<(int x, int y)> targets, GameBoard board)
255 {
256 return x >= 0 && x < board.Size && y >= 0 && y < board.Size &&
257 (board.Grid[x, y] == CellState.Empty || board.Grid[x, y] == CellState.Ship) &&
258 targets.Contains((x, y));
259 }

268 private List<(int x, int y)> GetNeighbors(int x, int y, GameBoard board)
269 {
270 var neighbors = new List<(int x, int y)>();
271 int[][] directions = { [-1, 0], [1, 0], [0, -1], [0, 1] }; // 4 стороны
272
273 foreach (var dir in directions)
274 {
275 int nx = x + dir[0];
276 int ny = y + dir[1];
277 if (nx >= 0 && nx < board.Size && ny >= 0 && ny < board.Size)
278 neighbors.Add((nx, ny));
279 }
280
281 return neighbors;
282 }
