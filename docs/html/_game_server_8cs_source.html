128 private ClientMessage? ParseMessage(string line)
129 {
130 var parts = line.Split(':', 2);
131 if (parts.Length < 2) return null;
132 var message = new ClientMessage { Type = parts[0] };
133 var dataPart = parts[1];
134 if (!string.IsNullOrEmpty(dataPart))
135 {
136 var dataPairs = dataPart.Split(';');
137 foreach (var pair in dataPairs)
138 {
139 var keyValue = pair.Split('=', 2);
140 if (keyValue.Length == 2)
141 {
142 message.Data[keyValue[0]] = keyValue[1];
143 }
144 }
145 }
146 return message;
147 }

155 private async Task ProcessMessageAsync(PlayerConnection player, ClientMessage message)
156 {
157 switch (message.Type.ToUpper())
158 {
159 case NetworkProtocol.Commands.Join:
160 await HandleJoinAsync(player, message.Data.GetValueOrDefault(NetworkProtocol.Keys.Name, "Anon"));
161 break;
162 case NetworkProtocol.Commands.PlaceShipsRandomly:
163 case NetworkProtocol.Commands.ShipPlaced:
164 break;
165 case NetworkProtocol.Commands.ShipPlacement:
166 await HandleShipPlacementAsync(player, message.Data);
167 break;
168 case NetworkProtocol.Commands.AllShipsPlaced:
169 await HandleShipsPlacedAsync(player);
170 break;
171 case NetworkProtocol.Commands.Attack:
172 await HandleAttackAsync(player, message.Data);
173 break;
174 case NetworkProtocol.Commands.LeaveGame:
175 await HandleLeaveGameAsync(player);
176 break;
177 case NetworkProtocol.Commands.ChatMessage:
178 await HandleChatMessageAsync(player, message.Data);
179 break;
180 case NetworkProtocol.Commands.ListReq:
181 await HandleList(player.Id);
182 break;
183 default:
184 Console.WriteLine($"[SERVER] Неизвестная команда от {player.Name} ({player.Id}): {message.Type}");
185 await SendServerMessageAsync(player, new ServerMessage
186 {
187 Type = NetworkProtocol.Commands.Error,
188 Data = { { NetworkProtocol.Keys.Message, $"[SERVER] Неизвестная команда: {message.Type}" } }
189 });
190 break;

415 private async Task HandleAttackAsync(PlayerConnection attacker, Dictionary<string, string> data)
416 {
417 var room = _rooms.Values.FirstOrDefault(r => r.Player1.Id == attacker.Id || r.Player2.Id == attacker.Id);
418 if (room == null || !room.GameStarted || room.GameOver)
419 {
420 Console.WriteLine($"[SERVER] Атака от {attacker.Name} отклонена: комната не найдена или игра не активна");
421 return;
422 }
423
424 if ((room.Player1.Id == attacker.Id && !room.Player1.IsMyTurn) ||
425 (room.Player2.Id == attacker.Id && !room.Player2.IsMyTurn))
426 {
427 await SendServerMessageAsync(attacker, new ServerMessage
428 {
429 Type = NetworkProtocol.Commands.Error,
430 Data = { { NetworkProtocol.Keys.Message, "Not your turn" } }
431 });
432 Console.WriteLine($"[SERVER] Атака от {attacker.Name} отклонена: не его ход");
433 return;
434 }
435
436 if (!int.TryParse(data.GetValueOrDefault("x", ""), out int x) ||
437 !int.TryParse(data.GetValueOrDefault("y", ""), out int y))
438 {
439 await SendServerMessageAsync(attacker, new ServerMessage
440 {
441 Type = NetworkProtocol.Commands.Error,
442 Data = { { NetworkProtocol.Keys.Message, "Invalid coordinates" } }
443 });
444 return;
445 }
446
447 var defender = room.GetOpponent(attacker);
448 if (defender == null)
449 {
450 Console.WriteLine($"[SERVER] Ошибка: не найден противник для {attacker.Name}");
451 return;
452 }
453 var defenderBoard = defender.Board;
454
455 // Проверяем, можно ли атаковать эту клетку
456 var cellState = defenderBoard.Grid[x, y];
457 if (cellState == CellState.Hit || cellState == CellState.Miss ||
458 cellState == CellState.Sunk || cellState == CellState.Blocked)
459 {
460 await SendServerMessageAsync(attacker, new ServerMessage
461 {
462 Type = NetworkProtocol.Commands.Error,
463 Data = { { NetworkProtocol.Keys.Message, "Cell already attacked" } }
464 });
465 Console.WriteLine($"[SERVER] Атака от {attacker.Name} по ({x},{y}) отклонена: клетка уже атакована (состояние={cellState})");
466 return;
467 }
468
469 Console.WriteLine($"[SERVER] Атака от {attacker.Name} по ({x},{y}), состояние клетки до атаки: {cellState}");
470
471 // Выполняем атаку
472 var (hit, sunk, gameOver) = defenderBoard.Attack(x, y);
473
474 Console.WriteLine($"[SERVER] Результат атаки: попадание={hit}, потоплен={sunk}, конец игры={gameOver}");
475
476 // Отправляем результаты атаки обоим игрокам
477 await SendAttackResultAsync(attacker, defender, x, y, hit, sunk, gameOver, defenderBoard);
478
479 // Обрабатываем переход хода или завершение игры
480 await HandleTurnTransitionAsync(room, attacker, defender, hit, gameOver);
481 }

495 private async Task SendAttackResultAsync(PlayerConnection attacker, PlayerConnection defender, int x, int y, bool hit, bool sunk, bool gameOver, GameBoard board)
496 {
497 var resultData = new Dictionary<string, string>
498 {
499 { NetworkProtocol.Keys.X, x.ToString() },
500 { NetworkProtocol.Keys.Y, y.ToString() },
501 { NetworkProtocol.Keys.Hit, hit.ToString().ToLower() },
502 { NetworkProtocol.Keys.Sunk, sunk.ToString().ToLower() },
503 { NetworkProtocol.Keys.GameOver, gameOver.ToString().ToLower() },
504 { NetworkProtocol.Keys.AttackerId, attacker.Id }
505 };
506
507 // Если корабль потоплен, отправляем информацию о заблокированных клетках
508 if (sunk)
509 {
510 var blockedCells = new List<string>();
511
512 // Находим потопленный корабль
513 var sunkShip = board.Ships.FirstOrDefault(s =>
514 s.IsSunk && s.Positions.Any(p => p.X == x && p.Y == y));
515
516 if (sunkShip != null)
517 {
518 Console.WriteLine($"[SERVER] Корабль потоплен! Позиции корабля: {string.Join(", ", sunkShip.Positions.Select(p => $"({p.X},{p.Y})"))}");
519
520 // Собираем все заблокированные клетки вокруг корабля
521 foreach (var pos in sunkShip.Positions)
522 {
523 for (int dx = -1; dx <= 1; dx++)
524 {
525 for (int dy = -1; dy <= 1; dy++)
526 {
527 int checkX = pos.X + dx;
528 int checkY = pos.Y + dy;
529
530 if (checkX >= 0 && checkX < board.Size &&
531 checkY >= 0 && checkY < board.Size)
532 {
533 if (board.Grid[checkX, checkY] == CellState.Blocked)
534 {
535 blockedCells.Add($"{checkX}:{checkY}");
536 }
537 }
538 }
539 }
540 }
541
542 // Добавляем позиции потопленного корабля
543 resultData[NetworkProtocol.Keys.SunkShipPositions] = string.Join(",",
544 sunkShip.Positions.Select(p => $"{p.X}:{p.Y}"));
545
546 Console.WriteLine($"[SERVER] Заблокировано клеток: {blockedCells.Count}");
547 }
548
549 if (blockedCells.Count > 0)
550 {
551 resultData[NetworkProtocol.Keys.BlockedCells] = string.Join(",", blockedCells);
552 }
553 }
554
555 var resultMsg = new ServerMessage
556 {
557 Type = NetworkProtocol.Commands.AttackResult,
558 Data = resultData
559 };
560
561 await SendServerMessageAsync(attacker, resultMsg);
562 await SendServerMessageAsync(defender, resultMsg);
563 }

574 private async Task HandleTurnTransitionAsync(GameRoom room, PlayerConnection attacker, PlayerConnection defender, bool hit, bool gameOver)
575 {
576 if (gameOver)
577 {
578 room.GameOver = true;
579 var winMsg = new ServerMessage
580 {
581 Type = NetworkProtocol.Commands.GameOver,
582 Data = { { NetworkProtocol.Keys.Winner, attacker.Name } }
583 };
584 await SendServerMessageAsync(attacker, winMsg);
585 await SendServerMessageAsync(defender, winMsg);
586 Console.WriteLine($"[SERVER] Игра в комнате {room.Id} завершена. Победитель: {attacker.Name}");
587 _rooms.TryRemove(room.Id, out _);
588 }
589 else if (!hit) // Промах - ход переходит
590 {
591 attacker.IsMyTurn = false;
592 defender.IsMyTurn = true;

594 await SendServerMessageAsync(attacker, new ServerMessage { Type = NetworkProtocol.Commands.OpponentTurn });
595 Console.WriteLine($"[SERVER] Ход переходит к {defender.Name}");
596 }
597 else // Попадание - ход остается
598 {
599 await SendServerMessageAsync(attacker, new ServerMessage { Type = NetworkProtocol.Commands.YourTurnAgain });
600 Console.WriteLine($"[SERVER] Попадание! {attacker.Name} стреляет снова");
