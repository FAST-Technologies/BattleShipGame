152 public (bool hit, bool sunk, bool gameOver) Attack(int x, int y)
153 {
154 // Проверка валидности координат
155 if (x < 0 || x >= Size || y < 0 || y >= Size)
156 return (false, false, false);
157
158 // Проверка, что клетка еще не атакована
159 if (Grid[x, y] == CellState.Hit || Grid[x, y] == CellState.Miss ||
160 Grid[x, y] == CellState.Sunk || Grid[x, y] == CellState.Blocked)
161 return (false, false, false);
162
163 if (Grid[x, y] == CellState.Ship)
164 {
165 Grid[x, y] = CellState.Hit;
166
167 var ship = Ships.FirstOrDefault(s => s.Positions.Contains((x, y)));
168 if (ship != null)
169 {
170 ship.HitCount++;
171 if (ship.IsSunk)
172 {
173 foreach (var pos in ship.Positions)
174 Grid[pos.X, pos.Y] = CellState.Sunk;
175 BlockAroundShip(ship);
176
177 bool gameOver = Ships.All(s => s.IsSunk);
178 return (true, true, gameOver);
179 }
180 }
181
182 return (true, false, false);
183 }
184 else
185 {
186 Grid[x, y] = CellState.Miss;
