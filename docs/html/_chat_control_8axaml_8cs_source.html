178 private string FormatTimestamp(DateTime timestamp)
179 {
180 var now = DateTime.Now;
181
182 if (timestamp.Date == now.Date)
183 {
184 return timestamp.ToString("HH:mm:ss");
185 }
186 else if (timestamp.Date == now.Date.AddDays(-1))
187 {
188 return "Вчера " + timestamp.ToString("HH:mm");
189 }
190 else if ((now - timestamp).TotalDays < 7)
191 {
192 return timestamp.ToString("dddd HH:mm", new System.Globalization.CultureInfo("ru-RU"));
193 }
194 else
195 {
196 return timestamp.ToString("dd.MM.yyyy HH:mm");
197 }
