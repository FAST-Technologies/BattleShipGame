103 private static void PlayTone(double frequency, int durationMs)
104 {
105 Task.Run(() =>
106 {
107 try
108 {
109 int sampleRate = 44100;
110 double amplitude = 0.5;
111
112 // Создаем WAV файл в памяти
113 using (var memoryStream = new System.IO.MemoryStream())
114 {
115 // Заголовок WAV файла
116 WriteWavHeader(memoryStream, durationMs, sampleRate);
117
118 // Аудиоданные (синусоида)
119 short[] data = new short[(int)(sampleRate * durationMs / 1000.0)];
120 double freq = frequency * 2.0 * Math.PI / sampleRate;
121
122 for (int i = 0; i < data.Length; i++)
123 {
124 data[i] = (short)(amplitude * Math.Sin(freq * i) * short.MaxValue);
125 }
126
127 // Конвертируем short[] в byte[]
128 byte[] byteData = new byte[data.Length * 2];
129 Buffer.BlockCopy(data, 0, byteData, 0, byteData.Length);
130
131 memoryStream.Write(byteData, 0, byteData.Length);
132 memoryStream.Position = 0;
133
134 // Воспроизводим
135 using (var audioFile = new WaveFileReader(memoryStream))
136 using (var outputDevice = new WaveOutEvent())
137 {
138 outputDevice.Init(audioFile);
139 outputDevice.Play();
140
141 // Ждем окончания звука
142 while (outputDevice.PlaybackState == PlaybackState.Playing)
143 {
144 Task.Delay(50).Wait();
145 }
146 }
147 }
148 }
149 catch (Exception ex)
150 {
151 Console.WriteLine($"[Sound Error] {ex.Message}");
152 }
153 });
154 }

165 private static void WriteWavHeader(System.IO.Stream stream, int durationMs, int sampleRate)
166 {
167 int numChannels = 1;
168 int bitsPerSample = 16;
169 int numSamples = sampleRate * durationMs / 1000;
170 int subChunk2Size = numSamples * numChannels * bitsPerSample / 8;
171
172 // RIFF заголовок
173 WriteString(stream, "RIFF");
174 WriteInt(stream, 36 + subChunk2Size);
175 WriteString(stream, "WAVE");
176
177 // fmt подзаголовок
178 WriteString(stream, "fmt ");
179 WriteInt(stream, 16);
180 WriteShort(stream, 1); // AudioFormat = PCM
181 WriteShort(stream, (short)numChannels);
182 WriteInt(stream, sampleRate);
183 WriteInt(stream, sampleRate * numChannels * bitsPerSample / 8); // ByteRate
184 WriteShort(stream, (short)(numChannels * bitsPerSample / 8)); // BlockAlign
185 WriteShort(stream, (short)bitsPerSample);
186
187 // data подзаголовок
188 WriteString(stream, "data");
189 WriteInt(stream, subChunk2Size);
190 }
